import random
import matplotlib.pyplot as plt

random.seed(0)
students = [ random.randint(80, 100) for i in range(30)]

scoreCounts = {}
for score in students:
    if score in scoreCounts:
        scoreCounts[score] += 1
    else:
        scoreCounts[score] = 1

print(scoreCounts)

x = [item[0] for item in scoreCounts.items()]
y = [item[1] for item in scoreCounts.items()]

plt.bar(x, y)
plt.show()
