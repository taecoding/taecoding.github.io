s07.1 Generation

Dictionary literals

>>> eng2sp = {"one": "uno", "two": "dos", "three": "tres"}

>>> eng2sp = {}
>>> eng2sp["one"] = "uno"
>>> eng2sp["two"] = "dos"

>>> print(eng2sp)
{"two": "dos", "one": "uno"}

From a list of tuples

d = dict([("two", "dos"), ("one": "uno)])

From two lists

names = ["Ryan", "Jack", "Nack"] ages = [15, 20, 18]

print(list(zip(names, ages))) print(dict(zip(names, ages)))

s07.2 Dictionary Operator

>>> inventory = {"apples": 430, "bananas": 312, "oranges": 525, "pears": 217}
>>> print(inventory)
{'pears': 217, 'apples': 430, 'oranges': 525, 'bananas': 312}

len

>>> len(inventory)

[]

>>> inventory["pears"] = 0
>>> print(inventory)
{'pears': 0, 'apples': 430, 'oranges': 525, 'bananas': 312}

in or not in

>>> "apples" in inventory
True
>>> "tomato" in inventory
False
>>> "tomato" not in inventory    
True

delete

>>> del inventory["pears"]
>>> print(inventory)
{'apples': 430, 'oranges': 525, 'bananas': 312}

s07.3 Dictionary methods

keys

for k in eng2sp.keys():   # The order of the k's is not defined
   print("Got key", k, "which maps to value", eng2sp[k])

ks = list(eng2sp.keys())
print(ks)

values

>>> list(eng2sp.values())
['tres', 'dos', 'uno']

items

>>> list(eng2sp.items())
[('three', 'tres'), ('two', 'dos'), ('one', 'uno')]

s07.4 Aliasing and copying

>>> opposites = {"up": "down", "right": "wrong", "yes": "no"}
>>> alias = opposites
>>> copy = opposites.copy()  # Shallow copy

>>> alias["right"] = "left"
>>> opposites["right"]
'left'

>>> copy["right"] = "privilege"
>>> opposites["right"]
'left'
