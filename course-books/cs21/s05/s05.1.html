s05.1 for loop

Range: start, stop, step

# sequence
for i in range(4):
    print('hello', i)

# list item
friends = ['Ryan', 'James', 'Peter']
for li in friends:
    print(li)

# index and item
for i, li in enumerate(friends):
    print(i, li)
