4.2 views

hw_app.views

from django.http import HttpRequest
from hw_app.models import Assignment, Question

def handle_assignments(request: HttpRequest):
    assignments = Assignment.objects.all()
    return render(request, 'assignments.html', 
        context={'assignments': assignments})


def handle_assignment_detail(request: HttpRequest, assignment_pk):
    assignment = Assignment.objects.filter(pk=assignment_pk)
    questions = Question.objects.filter(assignments_in=[assignment_pk])
    context = {
        "assignment": assignment,
        "questions": questions
    }
    return render(request, 'assignment_detail.html', context=context)
