Exercises: Binary Search

  1. write a function called binarySearch that search a target on array.

     def binarySearch(arr, target):
 low = 0
 high = len(arr) -1

 while (low <= high):
     mid = (low + high) // 2

     if arr[mid] == target:
         return mid
     elif arr[mid] > target:
         high = mid-1
     else:
         low = mid+1

 return -1

 print(binarySearch([2,5,8,12,16,23,38,56,72,91], 18))

def binaryClosestUpBar(items, bar):
    if (items[-1] < bar):
        return -1

    low = 0
    high = len(items)

    while low < high:
        mid = (low + high) // 2

        if items[mid] >= bar:
            high = mid
        else:
            low = mid + 1

    return low

print(binaryClosestUpBar([2,5,8,12,16,23,38,56,72,91], 24))
