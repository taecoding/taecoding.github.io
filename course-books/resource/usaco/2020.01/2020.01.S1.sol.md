#include &lt;algorithm&gt;
using namespace std;
 
int N,K;
int A[100000];
int mod;
 
bool cmp(int a,int b)
{
	return (a%mod) &gt; (b%mod);
}
 
int main()
{
	freopen("berries.in","r",stdin);
	freopen("berries.out","w",stdout);
	cin &gt;&gt; N &gt;&gt; K;
	int M = 0;
	for(int i=0;i&lt;N;i++)
	{
		cin &gt;&gt; A[i];
		M = max(M, A[i]);
	}
	int best = 0;
	for(int b=1;b &lt;= M;b++)
	{
		int full = 0;
		for(int i=0;i&lt;N;i++)
			full += A[i]/b;
		if(full &lt; K/2)
			break;
		if(full &gt;= K)
		{
			best = max(best, b*(K/2));
			continue;
		}
		mod = b;
		sort(A, A+N, cmp);
		int cur = b*(full - K/2);
		for(int i=0;i&lt;N &amp;&amp; i+full&lt;K;i++)
			cur += A[i]%b;
		best = max(best,cur);
	}
	cout &lt;&lt; best &lt;&lt; '\n';
}
