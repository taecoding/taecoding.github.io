3.2 s6/views.py

def handle_student_new(request: HttpRequest):
    if request.method == "GET":
        form = StudentForm()
    elif request.method == 'POST':
        form = StudentForm(request.POST)
        if form.is_valid():
            student = form.save()
            return redirect('students')

    return render(request, 's6/student_new.html',
                  context={'form': form})

4. student_edit

4.1 s6/views.py

def handle_student_edit(request: HttpRequest, student_pk):
    student = Student.objects.get(pk=student_pk)
    if request.method == 'GET':
        form = StudentForm(instance=student)
    elif request.method == 'POST':
        form = StudentForm(request.POST, instance=student)
        if form.is_valid():
            student = form.save()
            return redirect('students')

    return render(request, 's6/student_edit.html',
                  context={'student': student, 'form': form})

5. student_delete

5.1 s6/views.html

def handle_student_delete(request: HttpRequest, student_pk):
    student = Student.objects.get(pk=student_pk)
    if request.method == 'POST':
        student.delete()
        return redirect('students')

    return render(request, 's6/student_delete.html',
                  context={'student': student})
