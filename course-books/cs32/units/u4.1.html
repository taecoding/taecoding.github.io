WHERE

syntax

SELECT
 column_list
FROM
 table

 search_condition;

Condition

WHERE column_1 = 100;
WHERE column_1 > 100;
WHERE column_2 IN (1,2,3);
WHERE column_3 LIKE 'An%';
WHERE column_4 BETWEEN 10 AND 20;
  • AND returns 1 if both expressions are 1
  • OR returns true if either expression is 1
  • NOT reverses the value such as NOT EXISTS, NOT IN, NOT BETWEEN, etc.
  • IN returns 1 if a value is in a list of values.
  • LIKE returns 1 if a value matches a pattern
  • BETWEEN returns 1 if a value is within a range.
  • EXISTS returns 1 if a subquery contains any rows.

Inner join

Syntax

SELECT a1, a2, b1, b2
FROM 
    A
    INNER JOIN B 
        on B.f = A.f;
  • display trackid, name, title of album
    SELECT
t.trackid,
t.name,
a.title
FROM 
  tracks as t
  INNER JOIN albums as a 
      ON a.albumid = t.albumid;


SELECT
 trackid,
 tr.name AS Track,
 al.title AS Album,
 at.name AS Artist
FROM
 tracks as tr
 INNER JOIN albums as al
     ON al.albumid = tr.albumid
 INNER JOIN artists as at 
     ON at.artistid = al.artistid;

Group By

  • The GROUP BY clause returns one row for each group
  • aggregate function such as MIN, MAX, SUM, COUNT, or AVG
SELECT
 t.albumid,
 COUNT(t.trackid)
FROM
 tracks as t
GROUP BY
 t. albumid;

SELECT
 al.albumid,
 al.title,
 COUNT(tr.trackid)
FROM
 tracks as tr
 INNER JOIN albums as al 
     ON al.albumid = tr.albumid
GROUP BY
 al.albumid
HAVING COUNT(tr.trackid) > 15;
